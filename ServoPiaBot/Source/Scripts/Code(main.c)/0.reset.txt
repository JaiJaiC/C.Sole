/* =======================================================================
 * 🎯 你的校准数据
 * ======================================================================= */
// 加了 static 消除缺少声明的 warning
static int servo_offset[64] = {
     0,  0, -4,  0,  2, -3,  5,  3, -3, -2,  0, -5,  0, -3,  0, -1,
     2, -3,  0,  2, -5,  0,  0,  2,  2,  4,  0, -2, -5,  0,  0,  4,
     0,  0,  5,  0,  0,  0,  0, -2,  0,  0, -2,  0, -6,  0, -3, -2,
    -5,  0,  0,  0,  0,  0,  0,  0,  0,  0, -2,  0,  0,  0,  0, -2
};

static int servo_press_target[64] = {
    25, 40, 30, 45, 30, 45, 40, 30, 40, 30, 40, 30, 30, 40, 30, 40,
    30, 30, 40, 30, 40, 30, 40, 30, 30, 40, 30, 40, 30, 30, 40, 30,
    40, 30, 40, 30, 30, 40, 30, 40, 30, 30, 40, 30, 40, 30, 40, 30,
    30, 40, 30, 40, 30, 30, 40, 30, 40, 30, 40, 30, 30, 40, 30, 40
};

/* =======================================================================
 * 🎵 音乐核心：结构体定义
 * ======================================================================= */
typedef struct {
    uint8_t note_id;      
    uint16_t duration;    
    uint16_t delay_next;  
} NoteEvent;

/* =======================================================================
 * 👇👇👇👇 请把 PYTHON 脚本生成的数组粘贴在这里 👇👇👇👇
 * ======================================================================= */
// 这是一个占位用的示例曲子，防止你还没贴代码就编译报错
NoteEvent auto_song[] = {
    {24, 300, 100}, // Do
    {24, 300, 100}, // Do
    {31, 300, 100}, // Sol
    {31, 300, 100}, // Sol
    {255,  0, 500}  // 休止符
};
uint16_t song_length = sizeof(auto_song) / sizeof(NoteEvent);
/* =======================================================================
 * 👆👆👆👆 把你的曲子覆盖上面这段 auto_song 👆👆👆👆
 * ======================================================================= */

/* =======================================================================
 * 驱动与控制函数
 * ======================================================================= */
// 加上 static 和强制类型转换，彻底消灭所有 Warning
static void set_servo_angle(uint8_t id, uint8_t angle) {
    uint32_t pulse = 500 + ((uint32_t)angle * 2000 / 180);
    pca9685_set_mk(id, (int)pulse); 
}

static void play_song(NoteEvent* song, uint16_t length) {
    for(int i = 0; i < length; i++) {
        uint8_t current_note = song[i].note_id;
        
        if(current_note == 255) {
            HAL_Delay(song[i].duration + song[i].delay_next);
            continue;
        }

        // 加了 (uint8_t) 强转，消除丢失精度的 Warning
        set_servo_angle(current_note, (uint8_t)servo_press_target[current_note]);
        HAL_Delay(song[i].duration); 
        
        set_servo_angle(current_note, (uint8_t)(90 + servo_offset[current_note]));
        HAL_Delay(song[i].delay_next); 
    }
}

static void homing_all_servos(void) {
    for(int i = 0; i < 64; i++) {
        // 加了 (uint8_t) 强转，消除 Warning
        set_servo_angle((uint8_t)i, (uint8_t)(90 + servo_offset[i])); 
        HAL_Delay(15); 
    }
    HAL_Delay(1000); 
}

/* =======================================================================
 * 🚀 主函数
 * ======================================================================= */
int main(void)
{
    HAL_Init();
    __HAL_RCC_GPIOD_CLK_ENABLE();
    pca9685_init();

    // 开机强制平滑回位
    homing_all_servos();

    while(1)
    {
        play_song(auto_song, song_length);
        HAL_Delay(3000); 
    }
}
