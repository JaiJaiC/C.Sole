/* Includes ------------------------------------------------------------------*/
#include "main.h"
#include "stm32f4xx_hal.h"
#include "gpio.h"

#include "common.h"
#include "i2c_driver.h"
#include "pca9685_control.h"

/* =======================================================================
 * 🎯 你的校准数据 (绝对不要动这里，这是这台机器的灵魂)
 * ======================================================================= */
int servo_offset[64] = {
     0,  0, -4,  0,  2, -3,  5,  3, -3, -2,  0, -5,  0, -3,  0, -1,
     2, -3,  0,  2, -5,  0,  0,  2,  2,  4,  0, -2, -5,  0,  0,  4,
     0,  0,  5,  0,  0,  0,  0, -2,  0,  0, -2,  0, -6,  0, -3, -2,
    -5,  0,  0,  0,  0,  0,  0,  0,  0,  0, -2,  0,  0,  0,  0, -2
};

int servo_press_target[64] = {
    25, 40, 30, 45, 30, 45, 40, 30, 40, 30, 40, 30, 30, 40, 30, 40,
    30, 30, 40, 30, 40, 30, 40, 30, 30, 40, 30, 40, 30, 30, 40, 30,
    40, 30, 40, 30, 30, 40, 30, 40, 30, 30, 40, 30, 40, 30, 40, 30,
    30, 40, 30, 40, 30, 30, 40, 30, 40, 30, 40, 30, 30, 40, 30, 40
};

/* =======================================================================
 * 🎵 音乐核心引擎
 * ======================================================================= */
// 定义音符结构体
typedef struct {
    uint8_t note_id;      // 琴键编号 (0-63)，255 代表休止符
    uint16_t duration;    // 按下保持的时间 (毫秒)
    uint16_t delay_next;  // 抬起后，等多久再弹下一个音 (毫秒)
} NoteEvent;

// 🎹 《小星星》乐谱数组
// 假设：第 24 号舵机是 C调的 Do (C4)，那么：
// 24=Do, 26=Re, 28=Mi, 29=Fa, 31=Sol, 33=La (注意避开黑键)
NoteEvent twinkle_star[] = {
    {24, 300, 100}, // 1 (Do)   - 按下300ms，抬起等100ms
    {24, 300, 100}, // 1 (Do)
    {31, 300, 100}, // 5 (Sol)
    {31, 300, 100}, // 5 (Sol)
    {33, 300, 100}, // 6 (La)
    {33, 300, 100}, // 6 (La)
    {31, 600, 200}, // 5 (Sol)  - 长音，按久一点

    {29, 300, 100}, // 4 (Fa)
    {29, 300, 100}, // 4 (Fa)
    {28, 300, 100}, // 3 (Mi)
    {28, 300, 100}, // 3 (Mi)
    {26, 300, 100}, // 2 (Re)
    {26, 300, 100}, // 2 (Re)
    {24, 600, 800}  // 1 (Do)   - 结束句长音，多等一会儿
};

// 计算这首歌有多少个音符 (自动计算，不用手查)
uint16_t song_length = sizeof(twinkle_star) / sizeof(NoteEvent);

// 底层舵机驱动
void set_servo_angle(uint8_t id, uint8_t angle) {
    uint32_t pulse = 500 + (angle * 2000 / 180);
    pca9685_set_mk(id, pulse);
}

// 自动演奏函数
void play_song(NoteEvent* song, uint16_t length) {
    for(int i = 0; i < length; i++) {
        uint8_t current_note = song[i].note_id;
        
        // 如果是休止符 (编号 255)
        if(current_note == 255) {
            HAL_Delay(song[i].duration + song[i].delay_next);
            continue;
        }

        // 1. 按下！(读取专属深度表)
        set_servo_angle(current_note, servo_press_target[current_note]);
        HAL_Delay(song[i].duration); // 保持按压
        
        // 2. 抬起！(读取专属待命表)
        set_servo_angle(current_note, 90 + servo_offset[current_note]);
        HAL_Delay(song[i].delay_next); // 等待下个音符
    }
}

/* =======================================================================
 * 🚀 主函数
 * ======================================================================= */
int main(void)
{
    HAL_Init();
    __HAL_RCC_GPIOD_CLK_ENABLE();
    pca9685_init();

    /* 1. 开机安全归位 */
    for(int i = 0; i < 64; i++) {
        set_servo_angle(i, 90 + servo_offset[i]); 
        HAL_Delay(10);
    }
    HAL_Delay(2000); // 稳住 2 秒钟

    /* 2. 进入八音盒循环 */
    while(1)
    {
        // 呼叫演奏家，传入乐谱
        play_song(twinkle_star, song_length);
        
        // 弹完一首，谢幕 3 秒，然后重播
        HAL_Delay(3000); 
    }
}
